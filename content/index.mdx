{(() => {
  const showUntil = new Date('2025-12-19T23:59:59');
  const now = new Date();
  if (now <= showUntil) {
    return (
      <a href="https://banksalad.netlify.app/BAZDF" target="_blank" rel="noopener noreferrer">
        뱅크샐러드 대규모 채용중✨
      </a>
    );
  }
  return null;
})()}
